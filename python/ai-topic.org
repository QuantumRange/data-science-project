  bw = dict(bw)
  def score(text):
    # words = text.split()
    words = set(text.split())
    score = 0
    for w in words:
        if w in bw:
            score += bw[w]
    return score
#+end_src

#+begin_src jupyter-python
  import numpy as np
  pos_scores = np.zeros((len(df),))
  neg_scores = np.zeros((len(df2),))
  for i, t in enumerate(df['text']):
      pos_scores[i] = score(t)
      
  for i, t in enumerate(df2['text']):
      neg_scores[i] = score(t)

  print(pos_scores.mean(), pos_scores.min(), pos_scores.max(), np.median(pos_scores))
